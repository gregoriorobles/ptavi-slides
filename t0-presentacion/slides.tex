\section{Filosofía y Evaluación}


\begin{frame}
\frametitle{Evaluación}

\begin{itemize}
  \item Prácticas: hasta 5,5 puntos. \\ Mínimo: Parte básica práctica final.
  \item Teoría: hasta 5 puntos. \\ Mínimo: Superar la evaluación final de teoría.
  \item Al sumar teoría y prácticas, la suma ha de dar un 5 o más.
\end{itemize}


\end{frame}

\begin{frame}
\frametitle{Más cosas sobre la evaluación}

\begin{itemize}
  \item Convocatorias
  \begin{itemize}
    \item Si uno aprueba teoría o prácticas en enero, se ``guarda'' su nota para junio.
    \item No se ``guarda'' nota de un curso para otro.
  \end{itemize}
  \item Matrículas de honor. Algoritmo:
  \begin{itemize}
    \item Si sumas más de diez, obtienes un diez
    \item En caso de que haya más alumnos con nota superior a diez que matrículas de honor, se darán a aquéllos con la suma de notas más alta
    \item Al resto, les regalaremos un libro \emph{friqui} (dedicado, si quieres)
  \end{itemize}
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Evaluación Prácticas (I)}

\begin{itemize}
    \item Han ser ser realizadas de manera individual
\end{itemize}

\begin{itemize}
\item Evaluación continua (hasta 3,5 puntos)
  \begin{itemize}
    \item Las prácticas marcadas con un (*) se entregarán el lunes
          siguiente (hasta las 23:59) y darán hasta 0,7 puntos cada una.
  \end{itemize}
\item Práctica final (hasta 2 puntos)   
  \begin{itemize}
    \item Requisitos mínimos ({\bf obligatorios}) y adicionales (opcionales)
    \item Los requisitos mínimos puede que no sean iguales para todos
    \item Enunciado: Última clase de 2020. Entrega: Día antes del examen
    \item Seguirá habiendo tutorías en enero
  \end{itemize}
%\item \emph{Bonus} (hasta 0,5 puntos) por prácticas extraordinarias
\end{itemize}

\end{frame}

\begin{itemize}

%\item Según el cálculo que se realiza para ECTS, 6 créditos de los ``antiguos''

\item Cada ECTS supone una
dedicación de un estudiante ``medio'' de 25 horas por crédito.
\item Salen {\bf 150 horas} para la asignatura.
\item Presencialidad (50 horas):
\begin{itemize}
  \item 11 clases de teoría (+1 de presentación)
  \item 16 clases de prácticas
\end{itemize}
\item No presencialidad:
\begin{itemize}
  \item Preparación clase de teoría: 2 horas * clase de teoría
  \item Terminar prácticas a entregar: 6 horas * práctica
  \item Extra práctica final: 20 horas (7 de tutoría)
  \item Entrega práctica final: 1 hora
  \item Preparación evaluación: 5 horas
  \item Evaluación global final (teoría): 1 hora
\end{itemize}
\item Total = {\bf 135 horas}
\end{itemize}
